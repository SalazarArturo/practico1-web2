import { Usuarios } from "./../models/index.js";

async function findUserByEmail(email){
    //mantener el repository lo mas natural posible ... no maneja errores ni nada eso se lo delega al service
    const result = await Usuarios.findOne({where: {email}}); //que es exactamente result ? ... 
    return result;
   
}

async function createUser(userData){
    
    const userCreated = await Usuarios.create(userData);
    return userCreated;
}

export {
    findUserByEmail,
    createUser
}

que es exactamente result en nuestra funcion findUserByEmail? es una instancia de uno de nuestros modelos (tablas) hechos en sequelize

//sesiones nada que ver en el en el service eso se ve en el controller junto con todo lo que sea http, sesiones, etc

la property session de nuestro objeto request en express es un objeto que nos permite guardar informacion entre multiples request 
tal cual como lo veniamos haciendo antes con jwt para el token y guardar este token en una cookie para que el navegador lo almacene

la estructura de este objeto session es algo asi: 
req.session = {
    cookie: {
        path: '/',
        httpOnly: true,
        maxAge: 3600000
    },
    user: {
        id: 1,
        email: 'arturo@mail.com'
    }
}

primero necesitamos instalar la dependencia express-sesion para "activar" esta property del objeto req 
luego importamos la dependencia en nuestro index.js y la usamos como middleware global 

app.use(session({
    secret: process.env.SESSION_SECRET_WORD, //para firmar la cookie 
    resave: false, // si no hubo cambios en la sesion no sobreescribimos ni guardamos 
    saveUninitialized: false,
    cookie: {
        secure: false,
        maxAge: 1000 * 60 * 60
    }
}));

cuando llega un request la sesion es guardada en memoria y con el resave decimos que si no hubo cambios en la sesion en el proximo
request pa que vamos a guardar si es lo mismo ... ahorro de memoria paa 

y el saveUninitialized es para crear una sesion automaticamente aunque nosotros no lo hayamos hecho explicitamente 
por eso le damos valor false, si no lo especificamos pues es porque no es necesario sesion, lo mismo .. gestion de memoria 

entonces en este flujo de login tomamos el primer request del cliente y ahi creamos la sesion 
esto gatillara que guardamos la info de la sesion en memoria y se aliste una cookie en nuestra response para el cliente
el cliente en su proxima request devolvera esa cookie y ahi dentro tenemos la info de la sesion para corroborar datos y seguir con las operaciones
